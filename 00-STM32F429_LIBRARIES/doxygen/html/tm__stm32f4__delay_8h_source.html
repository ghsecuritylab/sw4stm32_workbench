183 static __INLINE void Delay(uint32_t micros) {
184 #if defined(TM_DELAY_TIM)
185  volatile uint32_t timer = TM_DELAY_TIM->CNT;
186 
187  do {
188  /* Count timer ticks */
189  while ((TM_DELAY_TIM->CNT - timer) == 0);
190 
191  /* Increase timer */
192  timer = TM_DELAY_TIM->CNT;
193 
194  /* Decrease microseconds */
195  } while (--micros);
196 #else
197  uint32_t amicros;
198 
199  /* Multiply micro seconds */
200  amicros = (micros) * (mult);
201 
202  #ifdef __GNUC__
203  if (SystemCoreClock == 180000000 || SystemCoreClock == 100000000) {
204  amicros -= mult;
205  }
206  #endif
207 
208  /* If clock is 100MHz, then add additional multiplier */
209  /* 100/3 = 33.3 = 33 and delay wouldn't be so accurate */
210  #if defined(STM32F411xE)
211  amicros += mult;
212  #endif
213 
214  /* While loop */
215  while (amicros--);
216 #endif /* TM_DELAY_TIM */
217 }
